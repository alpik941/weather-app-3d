💻 Быстрая интеграция

import RealisticSnowfall from './RealisticSnowfall';

const WeatherScene = ({ temperature }) => {
  return (
    <Canvas>
      <fog attach="fog" args={['#d0d8e0', 5, 45]} />
      
      <RealisticSnowfall
        count={600}
        intensity={1.0}
        windSpeed={0.15}
        renderMode={"detailed"}
        accumulation={temperature < -5}
      />
    </Canvas>
  );
};
